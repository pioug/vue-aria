# @vue-spectrum/s2

## Example

```ts
import { ref } from "vue";
import {
  ActionButton,
  ActionButtonGroup,
  Accordion,
  ActionMenu,
  Avatar,
  Badge,
  Button,
  ButtonGroup,
  ComboBox,
  ColorField,
  Checkbox,
  CheckboxGroup,
  CloseButton,
  DateField,
  Disclosure,
  DisclosurePanel,
  DisclosureTitle,
  Divider,
  Image,
  IllustratedMessage,
  InlineAlert,
  Link,
  Meter,
  NumberField,
  Provider,
  ProgressBar,
  ProgressCircle,
  SearchField,
  Radio,
  RadioGroup,
  SelectBox,
  SelectBoxGroup,
  Slider,
  StatusLight,
  Switch,
  TextField,
  ToggleButton,
  ToggleButtonGroup,
  Well,
  pressScale,
} from "@vue-spectrum/s2";
import { theme } from "@vue-spectrum/theme-default";

const elementRef = ref<HTMLElement | null>(null);
const resolveStyle = pressScale(elementRef, { willChange: "opacity" });
const style = resolveStyle({ isPressed: true });

const app = {
  components: { Provider, Button },
  setup() {
    return { theme };
  },
  template: `
    <Provider :theme="theme">
      <Accordion>
        <Disclosure id="section-a">
          <DisclosureTitle>Section A</DisclosureTitle>
          <DisclosurePanel>Panel A</DisclosurePanel>
        </Disclosure>
      </Accordion>
      <ActionButtonGroup size="L" isQuiet>
        <ActionButton>Edit</ActionButton>
        <ToggleButton>Pin</ToggleButton>
      </ActionButtonGroup>
      <Avatar src="https://example.com/avatar.png" alt="User avatar" />
      <Badge variant="positive">Ready</Badge>
      <ToggleButtonGroup size="S">
        <ToggleButton>Bold</ToggleButton>
        <ToggleButton>Italic</ToggleButton>
      </ToggleButtonGroup>
      <CloseButton />
      <ActionMenu :items="[{ key: 'edit', label: 'Edit' }]" />
      <SearchField label="Search items" />
      <ColorField label="Accent color" defaultValue="#1473e6" />
      <Checkbox>Subscribe</Checkbox>
      <CheckboxGroup aria-label="Pets" orientation="horizontal">
        <Checkbox value="dogs">Dogs</Checkbox>
        <Checkbox value="cats">Cats</Checkbox>
      </CheckboxGroup>
      <DateField label="Publish date" />
      <Image src="https://example.com/photo.png" alt="Landscape" />
      <IllustratedMessage>
        <h3>No results</h3>
        <p>Try a different query.</p>
      </IllustratedMessage>
      <InlineAlert variant="notice">
        <h3>Attention</h3>
        <p>Review required.</p>
      </InlineAlert>
      <Link href="https://example.com">Visit</Link>
      <NumberField aria-label="Quantity" />
      <ProgressBar label="Loading" :value="40" />
      <ProgressCircle aria-label="Loading circle" :value="75" />
      <Meter label="Usage" :value="66" variant="critical" />
      <Slider label="Volume" />
      <Switch aria-label="Notifications">Notifications</Switch>
      <TextField label="Title" />
      <RadioGroup aria-label="View mode" orientation="horizontal">
        <Radio value="list">List</Radio>
        <Radio value="grid">Grid</Radio>
      </RadioGroup>
      <ComboBox label="Category" :items="[{ key: 'a', label: 'A' }]" />
      <SelectBoxGroup>
        <SelectBox id="option1">Option 1</SelectBox>
        <SelectBox id="option2">Option 2</SelectBox>
      </SelectBoxGroup>
      <Divider />
      <StatusLight role="status" variant="notice">Syncing</StatusLight>
      <Well role="region">Helpful content</Well>
      <ButtonGroup>
        <Button variant="premium">Create</Button>
      </ButtonGroup>
    </Provider>
  `,
};
```

## Notes

- This baseline now includes first component primitives (`Provider`, `Accordion`, `Disclosure`, `DisclosurePanel`, `DisclosureTitle`, `ActionMenu`, `Button`, `LinkButton`, `ActionButton`, `ActionButtonGroup`, `Avatar`, `Badge`, `ButtonGroup`, `ComboBox`, `ColorField`, `Checkbox`, `CheckboxGroup`, `CloseButton`, `DateField`, `Divider`, `Image`, `IllustratedMessage`, `InlineAlert`, `Link`, `Meter`, `NumberField`, `ProgressBar`, `ProgressCircle`, `Radio`, `RadioGroup`, `SearchField`, `SelectBox`, `SelectBoxGroup`, `Slider`, `StatusLight`, `Switch`, `TextField`, `ToggleButton`, `ToggleButtonGroup`, `Well`) and shared utilities (`pressScale`, docs-env gate).
- `SelectBoxGroup` includes baseline keyboard parity for Space/Enter selection and Arrow/Home/End focus navigation (with single-select arrow navigation updating selection and disabled-option skipping).
- Full S2 component exports and parity remain in progress and are tracked in the Spectrum roadmap.
